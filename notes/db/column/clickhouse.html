SELECT table,
       name,
       sum(data_compressed_bytes)                  AS compressed,
       sum(data_uncompressed_bytes)                AS uncompressed,
       floor((compressed / uncompressed) * 100, 4) as percent
FROM system.columns
WHERE database = currentDatabase()
GROUP BY table,
         name
ORDER BY table ASC,
         name ASC;
